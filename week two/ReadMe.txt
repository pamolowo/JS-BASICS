ARITHMETIC OPERATORS
+, -, *, **, /, %, ++, --

EXAMPLES:
let a = 2;
let b = 3;

let c = a + b;
let result = a * b;

ASSIGNMENT OPERATORS
=, +=, -=, *=, /=, %=, **=

EXAMPLES:
let a = 2;
let b = 3;

let a += b;
x = x / y

COMPARISON OPERATORS
==,===,!=,!==,>,<,>=,<=,

EXAMPLES:
let x = 5;
let y = 10;
  x > 8;

  y <= 8;

LOGICAL OPERATORS
&&, ||, !
EXAMPLES:
let x = 6;
ley y = 3;
(x < 7 && y > 1) is true
(x == 3 || y == 3) is false

BITWISE OPERATORS:
&, |, ^, ~, <<, >>, >>>

EXAMPLES:
5 & 1;
5 >> 1;


4. 1
   8
   15
